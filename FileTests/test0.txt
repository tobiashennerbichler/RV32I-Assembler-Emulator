_start:
    ADDI sp, zero, 0x700
    JAL ra, main
    EBREAK

main:
    ADDI sp, sp, -16
    SW ra, 12(sp)
    SW fp, 8(sp)
    ADDI fp, sp, 16

    ADD t0, zero, zero
    SW t0, -12(fp)

    ADD t1, zero, zero
    SW t1, -16(fp)

    JAL ra, loop

    ADDI t2, zero, 5

loop:
    ADDI t0, t0, 3
    ADDI t1, t1, 1
    BLT t1, t2, loop

    LW ra, 12(sp)
    LW fp, 8(sp)
    ADDI sp, sp, 16
    JALR zero, 0(ra)
